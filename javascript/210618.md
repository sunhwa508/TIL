var isValid = function(s) {
  if (s === null || s.length <= 0) return true;
    var stack = [];
    for (var c of s) {
        console.log("c",c)
        if (c === '[') stack.push(']');
        else if (c === '{') stack.push('}');
        else if (c === '(') stack.push(')');
        else if (stack.length === 0 || c !== stack.pop()) return false;
    }
   return stack.length === 0;
};
